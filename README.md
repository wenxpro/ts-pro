初始化
```
npm init -y
```
安装 webpack
```
npm i -D webpack webpack-cli typescript ts-loader
```
使用webpack插件
```
npm i -D html-webpack-plugin
```
使用webpack开发服务
```
npm i -D webpack-dev-server
```
使用webpack清空插件
```
npm i -D clean-webpack-plugin
```
使用babel增加兼容性
```
npm i -D @babel/core @babel/preset-env babel-loader core-js
```
